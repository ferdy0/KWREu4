# Abolish slavery
country_event = {
	id = levantine_events.12
	title = levantine_events.12.t
	desc = levantine_events.12.d
	picture = BIG_BOOK_eventPicture

	fire_only_once = yes

	trigger = {
		religion = levantine
		has_global_modifier_value = {
        	which = cavalry_flanking
        	value = 9
        }
	}

	option = {
		name = levantine_events.12.a
		add_prestige = 5
        add_country_modifier = {
            name = innovative_modifier
            duration = 3650
        }
        add_country_modifier = {
            name = counter_trigger_levantine_cavflanking
            duration = -1
        }
        add_country_modifier = {
            name = "the_abolish_slavery_act"
            duration = -1
        }
        custom_tooltip = abolished_slavery_act_tooltip
        hidden_effect = {
            every_owned_province = {
        	    limit = {
        		    trade_goods = slaves
        		}
        		change_trade_goods = unknown
        		remove_building = tradecompany
        		if = {
        		    limit = { has_province_modifier = slave_entrepot }
        		    remove_province_modifier = slave_entrepot
        		}
        	}
        }
	}
}

country_event = {
	id = levantine_events.13
	title = levantine_events.13.t
	desc = levantine_events.13.d
	picture = ANGRY_MOB_eventPicture

	trigger = {
		religion = levantine
		NOT = { has_country_modifier = merchants_cowed_levantine }
		NOT = { has_country_flag = greed_is_sin }
	}

	mean_time_to_happen = {
		months = 240

		modifier = {
            factor = 2
            has_country_flag = usury_is_sin
        }
	}

	option = {
		name = levantine_events.13.a
		random_owned_province = {
		    add_province_modifier = {
                name = absent_merchants_levantine
                duration = 3650
            }
		}
		add_country_modifier = {
            name = merchants_cowed_levantine
            duration = 3650
        }
	}
}
