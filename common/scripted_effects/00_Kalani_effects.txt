kalani_release_finland_effect = {
	hidden_effect = {
		FIN = {
			every_core_province = { remove_core = FIN }
		}
		every_owned_province = {
			limit = {
				OR = {
					area = finland_area
					area = bothnia_area
					province_id = 30
					province_id = 32
					province_id = 4124
					province_id = 4151
					province_id = 1930
				}
			}
			add_core = FIN
			remove_core = ROOT 
		}
	}
	release = FIN
	hidden_effect = {
		if = { # Set capital in Abo
			limit = {
				27 = { owned_by = FIN }
			}
			FIN = {
				set_capital = 27
				if = {
					limit = { 27 = { religion = catholic } }
					change_religion = catholic
				}
				else_if = {
					limit = { 27 = { religion = protestant } }
					change_religion = protestant
				}
				else_if = {
					limit = { 27 = { religion = reformed } }
					change_religion = reformed
				}
				else = {
					change_religion = orthodox
				}
			}
		}
		else_if = { # Set capital in Helsinki as a fallback
			limit = {
				28 = { owned_by = FIN }
			}
			FIN = {
				set_capital = 28
				if = {
					limit = { 28 = { religion = catholic } }
					change_religion = catholic
				}
				else_if = {
					limit = { 28 = { religion = protestant } }
					change_religion = protestant
				}
				else_if = {
					limit = { 28 = { religion = reformed } }
					change_religion = reformed
				}
				else = {
					change_religion = orthodox
				}
			}
		}
		else = {
			FIN = {
				if = {
					limit = { capital_scope = { religion = catholic } }
					change_religion = catholic
				}
				else_if = {
					limit = { capital_scope = { religion = protestant } }
					change_religion = protestant
				}
				else_if = {
					limit = { capital_scope = { religion = reformed } }
					change_religion = reformed
				}
				else = {
					change_religion = orthodox
				}
			}
		}
		FIN = {
			change_primary_culture = finnish
			set_prestige_to_zero = yes
			add_adm_power = 50
			add_dip_power = 50
			add_mil_power = 50
		}
	}
}

kalani_german_colonist_culture_decider = {
	if = { # If you're North German, pick your Primary Culture
		limit = {
			OR = {
				primary_culture = pommeranian
				primary_culture = prussian
				primary_culture = baltic_german
				primary_culture = lower_saxon
				primary_culture = hannoverian
				primary_culture = hessian
				primary_culture = saxon
			}
		}
		save_event_target_as = german_colonist_coutry
	}
	else_if = { # Otherwise, grab a North German province of an accepted culture
		limit = {
			OR = {
				accepted_culture = pommeranian
				accepted_culture = prussian
				accepted_culture = baltic_german
				accepted_culture = lower_saxon
				accepted_culture = hannoverian
				accepted_culture = hessian
				accepted_culture = saxon
			}
			any_owned_province = {
				OR = {
					region = north_german_region
					region = baltic_region
				}
				has_owner_accepted_culture = yes
				OR = {
					culture = pommeranian
					culture = prussian
					culture = baltic_german
					culture = lower_saxon
					culture = hannoverian
					culture = hessian
					culture = saxon
				}
			}
		}
		random_owned_province = {
			limit = {
				has_owner_accepted_culture = yes
				OR = {
					culture = pommeranian
					culture = prussian
					culture = baltic_german
					culture = lower_saxon
					culture = hannoverian
					culture = hessian
					culture = saxon
				}
			}
			save_event_target_as = german_colonist_province
		}
	}
}

        }

                    NOT = { has_owner_accepted_culture = yes }

			}
			remove_province_modifier = nwc_cultural_autonomy_modifier

	if = {

			NOT = { accepted_culture = icelandic }
		}
