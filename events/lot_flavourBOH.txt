namespace = lot_flavor_boh
normal_or_historical_nations = yes

#Hussites march on Prague
country_event = {
	id = lot_flavor_boh.1
	title = lot_flavor_boh.1.t
	desc = lot_flavor_boh.1.d
	picture = {	
		trigger = {
			has_dlc = "Emperor"
		}
		picture = HUSSITE_eventPicture
	}
	picture = {	
		trigger = {
			NOT = { has_dlc = "Emperor" }
		}
		picture = CIVIL_WAR_eventPicture
	}

	fire_only_once = yes

	trigger = {
		OR = {
			tag = BOH
			was_tag = BOH
		}
		owns = 266
		NOT = {
			AND = {
				is_year = 1500
				has_ruler_flag = he_who_hath_made_compromise
			}
		}
		is_at_war = no
		OR = {
			religion = catholic
			religion = hussite
		}
		is_free_or_tributary_trigger = yes
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = first_hussite_king
			set_country_flag = no_election_of_new_ruler_event_trigger
			if = {
				limit = {
					ai = yes
					has_dlc = "Winds of Change"
				}
				country_event = { id = flavor_boh.14 }
			}
		}
	}
	after = {
		clr_country_flag = no_election_of_new_ruler_event_trigger
	}
	
	option = { # Jirí's Way
		ai_chance = { factor = 10 }
		name = lot_flavor_boh.1.a
		set_country_flag = hussite_regency_mission_flag
		if = {
			limit = {
				has_regency = no
			}
			tooltip = {
				kill_ruler = yes
			}
		}
		if = {
			limit = {
				has_heir = yes
			}
			remove_heir = {}
		}
		change_religion = hussite
		define_ruler = {
			name = "Jirí"
			dynasty = "z Podebrad"
			adm = 5
			dip = 4
			mil = 3
			religion = hussite
			culture = czech
			age = 25
		}
		hidden_effect = {
			set_ruler_flag = ruler_cant_trigger_this_event_flag
		}
		if = {
			limit = {
				has_states_general_mechanic = yes
			}
			change_statists_vs_orangists = 1
		}
		if = {
			limit = {
				has_dlc = "Rights of Man" 
			}
			hidden_effect = { add_ruler_personality_ancestor = { key = zealot } }
		}
		add_reform_desire = 0.02
	}
	
	option = { # Convert to Hussite
		ai_chance = { factor = 10 }
		name = lot_flavor_boh.1.b
		set_country_flag = hussite_regency_mission_flag
		change_religion = hussite
		set_ruler_religion = hussite
		set_heir_religion = hussite
		if = {
			limit = {
				has_states_general_mechanic = yes
			}
			change_statists_vs_orangists = 1
		}
		if = {
			limit = {
				has_dlc = "Rights of Man" 
			}
			hidden_effect = { add_ruler_personality_ancestor = { key = zealot } }
		}
		add_reform_desire = 0.02
		266 = {
			spawn_rebels = {
				type = pretender_rebels
				size = 1
				leader = "Jirí"
				leader_dynasty = "z Podebrad"
			}		
		}
	}
	
	option = { # Hussite King, Catholic nation - tolerance option
		ai_chance = { factor = 90 }
		set_country_flag = hussite_regency_mission_flag
		name = lot_flavor_boh.1.c
		add_papal_influence = -100
		set_ruler_religion = hussite
		set_heir_religion = hussite
		if = {
			limit = {
				has_states_general_mechanic = yes
			}
			change_statists_vs_orangists = 1
		}
		add_reform_desire = 0.01
		add_ruler_modifier = {
			name = "boh_hussite_king"
			duration = -1
		}
		PAP = {
			country_event = { id = flavor_boh.9 }
		}
	}
	
	option = { # Fight them
		name = flavor_boh.1.e
		set_country_flag = hussite_regency_mission_flag
		266 = {
			spawn_rebels = {
				type = pretender_rebels
				size = 1
				leader = "Jirí"
				leader_dynasty = "z Podebrad"
			}		
		}
		random_owned_province = {
			limit = {
				NOT = { province_id = 266 }
			}
			spawn_rebels = {
				type = pretender_rebels
				size = 1
			}
		}
		if = {
			limit = { religion = catholic }
			add_papal_influence = 10
			PAP = {
				add_opinion = {
					who = root
					modifier = boh_papal_reconciled
				}
			}
		}
	}
}
