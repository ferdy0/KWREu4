ming_crisis = {
	potential = {
		tag = XUE
		NOT = { has_country_flag = xue_crisis_happened }
		normal_or_historical_nations = yes
	}
	
	can_start = {
		has_any_disaster = no
		NOT = { current_age = age_of_discovery }
		had_country_flag = { flag = xue_dynasty_formed days = 27375 } # 75 years, as per Luna's request
		if = {
			limit = {	
				has_dlc = "Mandate of Heaven"
			}
			OR = {
				AND = {
					NOT = { imperial_mandate = 50 }
					is_emperor_of_china = yes
				}
				NOT = { is_emperor_of_china = yes }
			}
		}
		else = {
			OR = {
				has_country_modifier = the_mandate_of_heaven_lost
				has_country_modifier = lost_mandate_of_heaven
			}
		}
	}
	
	can_stop = {
		has_any_disaster = yes
		if = {
			limit = {
				has_dlc = "Mandate of Heaven"
			}
			imperial_mandate = 50
		}
		else = {
			NOT = { has_country_modifier = the_mandate_of_heaven_lost }
			NOT = { has_country_modifier = lost_mandate_of_heaven }
		}
	}
	
	progress = {
		modifier = {
			factor = 10
			has_dlc = "Mandate of Heaven"
			NOT = { is_emperor_of_china = yes }
		}
		modifier = {
			factor = 2
			NOT = { imperial_mandate = 50 }
			imperial_mandate = 30
		}
		modifier = {
			factor = 3
			NOT = { imperial_mandate = 30 }
			imperial_mandate = 10
		}
		modifier = {
			factor = 10
			NOT = { imperial_mandate = 10 }
		}
		modifier = {
			factor = 1
			num_of_loans = 5
		}
		modifier = {
			factor = 3
			num_of_loans = 10
		}
		modifier = {
			factor = 5
			is_bankrupt = yes
		}
		modifier = {
			factor = 2
			corruption = 5
			NOT = { corruption = 10 }
		}
		modifier = {
			factor = 5
			corruption = 10
		}
	}
	
	can_end = {
		if = {
			limit = {
				has_dlc = "Mandate of Heaven"
			}
			OR = {
				imperial_mandate = 75
				is_emperor_of_china = no
			}
		}
		else = {
			stability = 3
		}
		is_bankrupt = no
		NOT = { num_of_loans = 5 }
		NOT = { corruption = 5 }
		NOT = { num_of_rebel_armies = 1 }
		NOT = { num_of_rebel_controlled_provinces = 1 }
	}
	
	modifier = {
		land_morale = -0.15
		technology_cost = 0.2
		global_unrest = 15
		global_monthly_devastation = 0.08
		global_tax_modifier = -0.5
		global_trade_goods_size_modifier = -0.3
	}
	
	on_start = ming_crisis.1 # Potentially new event to give some flavor?
	on_end = ming_crisis.7 # Potentially new event to give some flavor?
	
	on_monthly = {
		events = {
			mch_missions.5 # Koxinga flees to Taiwan
			xue_disaster.1 # Southern dynasty rises up
			xue_disaster.2 # Northern dynasty rises up
			xue_disaster.3 # Jurchen Protectorate
		}
		random_events = {
			1000 = 0
			100 = ming_crisis.6 # Rebellion
			100 = xue_disaster.4
		}
	}
}
