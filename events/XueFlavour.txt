				province_id = 1833
				province_id = 2159
				province_id = 2161
				province_id = 2174
			}
		}
	}

	immediate = {
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
						controlled_by = REB
						controlled_by = ROOT
					}
					OR = {
						area = fujian_area
						area = guangdong_area
						area = guangxi_area
						province_id = 665
						province_id = 670
						province_id = 1833
						province_id = 2159
						province_id = 2161
						province_id = 2174
					}
				}
				add_core = HUA
				clear_rebels = yes
			}
			release = HUA
			if = { # Set the correct capital
				limit = {
					2158 = { owned_by = HUA }
				}
				HUA = { set_capital = 2158 }
			}
			else_if = {
				limit = {
					667 = { owned_by = HUA }
				}
				HUA = { set_capital = 667 }
			}
			HUA = {
				capital_scope = {
					build_to_forcelimit = {
						infantry = 0.6
						cavalry = 0.3
						artillery = 0.1
					}
				}
				change_government = monarchy
				add_government_reform = chinese_warlord
				adopt_reform_progress = ROOT
				define_ruler = {
					name = "Zicheng"
					dynasty = "Li"
				}
				add_yearly_manpower = 15
				add_years_of_income = 15
			}
			if = { # Definitely not the Taiping I promise guys
				limit = {
					HUA = { religion = confucianism }
				}
				if = {
					limit = {
						HUA = {
							any_owned_province = { religion = catholic }
						}
					}
					HUA = { add_harmonized_religion = catholic }
				}
				if = {
					limit = {
						HUA = {
							any_owned_province = { religion = protestant }
						}
					}
					HUA = { add_harmonized_religion = protestant }
				}
				if = {
					limit = {
						HUA = {
							any_owned_province = { religion = reformed }
						}
					}
					HUA = { add_harmonized_religion = reformed }
				}
			}
			if = {
				limit = {
					any_country = {
						NOT = { is_neighbor_of = ROOT }
						is_subject_of_type = tributary_state
						is_subject_of = ROOT
					}
				}
				every_country = {
					limit = {
						NOT = { is_neighbor_of = ROOT }
						is_subject_of_type = tributary_state
						is_subject_of = ROOT
					}
					country_event = {
						id = ming_crisis.10
					}
				}
			}
		}
	}

	option = {
		name = xue_disaster.1.a
		add_stability = -2
		add_mandate = -10
		if = {
			limit = {
				has_dlc = "Mandate of Heaven"
				is_emperor_of_china = yes
			}
			HUA = {
				declare_war_with_cb = {
					who = root
					casus_belli = cb_take_mandate
				}
			}
		}
		else = {
			HUA = {
				declare_war = root
			}
		}
	}
	
	option = {
		name = xue_disaster.1.b
		trigger = {
			ai = no
		}
		add_stability = -2
		add_mandate = -10
		if = {
			limit = {
				has_dlc = "Mandate of Heaven"
				is_emperor_of_china = yes
			}
			HUA = {
				declare_war_with_cb = {
					who = root
					casus_belli = cb_take_mandate
				}
			}
		}
		else = {
			HUA = {
				declare_war = root
			}
		}
		switch_tag = HUA
	}
}

		any_owned_province = {
			superregion = china_superregion

				NOT = { tag = ROOT }

				NOT = { tag = CSH } # Somehow... the Shun Dynasty has returned

			}

	immediate = {

						NOT = { tag = MNG }
						NOT = { tag = CSH }

					limit = {
						NOT = { tag = ROOT }

		}

			factor = 100
		}
	}
	
	option = {
		name = ming_crisis.11.b
		add_mandate = -5

				limit = {
					owned_by = ROOT
				}
				add_local_autonomy = 10
			}
		}
		ai_chance = {
			factor = 0
		}
	}
