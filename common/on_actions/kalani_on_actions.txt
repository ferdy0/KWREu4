    if = {
        limit = {
            owner = { has_country_flag = nwc_growing_pains_flag }
        }
        owner = {
            kalani_growing_pains_calculation_effect = yes
        }
    }
    else_if = {
        limit = {
            owner = { has_country_flag = nwc_mild_growing_pains_flag }

        }
    }
}
